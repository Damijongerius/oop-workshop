procedureel programmeren

  • focus op functies
  • Globale Variabelen
  • staps gewijs

							#include 

								int main() {
									int getallen[] = {1, 2, 3, 4, 5};
									int lengte = sizeof(getallen) / sizeof(getallen[0]);
									int som = 0;
								
									for (int i = 0; i < lengte; i++) {
										som += getallen[i] * getallen[i];
									}
								
									printf("De som van de kwadraten is: %d\n", som);
								
									return 0;
								}
